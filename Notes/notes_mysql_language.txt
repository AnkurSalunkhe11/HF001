   DDL (Data Definition Language): Defines or changes the structure of database objects (databases, tables, columns, indexes).
       Common: `CREATE`, `ALTER`, `DROP`, `TRUNCATE`, `RENAME`

   DML (Data Manipulation Language): Adds, changes, or removes the data inside tables.
       Common: `INSERT`, `UPDATE`, `DELETE`, `REPLACE`

   DQL (Data Query Language): Queries (reads) the data.
       Common: `SELECT`


   TCL (Transaction Control): `START TRANSACTION`, `COMMIT`, `ROLLBACK`, `SAVEPOINT`
   DCL (Access Control): `GRANT`, `REVOKE`



 1) DDL — Define/Change Structure

 1.1 CREATE (database & table)


-- Create a database
CREATE DATABASE company_db;

-- Use the database
USE company_db;

-- Create a table

CREATE TABLE table_name (
    column1 datatype [constraints],
    column2 datatype [constraints],
    ...
);

CREATE TABLE employees (
    id INT PRIMARY KEY AUTO_INCREMENT,
    first_name VARCHAR(50) NOT NULL,
    last_name  VARCHAR(50) NOT NULL,
    email      VARCHAR(100) UNIQUE,
    hire_date  DATE NOT NULL,
    salary     DECIMAL(10,2) DEFAULT 0.00
) 


 1.2 ALTER (add/modify/drop columns, indexes)


-- Add a column
ALTER TABLE table_name
ADD column_name datatype constraints;

ALTER TABLE employees
ADD department VARCHAR(50) DEFAULT 'General';

-- Modify a column type/constraint
ALTER TABLE table_name
MODIFY column_name new_datatype;

ALTER TABLE employees
MODIFY salary DECIMAL(12,2) NOT NULL;

-- Rename a column (MySQL 8.0+)
ALTER TABLE table_name
CHANGE old_column_name new_column_name datatype;

ALTER TABLE employees
RENAME COLUMN department TO dept;

-- Drop a coulumn
ALTER TABLE table_name
DROP COLUMN column_name;

-- Add an index


ALTER TABLE employees
ADD INDEX idx_last_name (last_name);


 1.3 RENAME (table)

ALTER TABLE old_table_name
RENAME TO new_table_name;

or

RENAME TABLE old_name TO new_name;

RENAME TABLE employee s TO staff;


--drop database 
DROP DATABASE database_name;

 1.4 TRUNCATE (remove all rows, keep structure; faster than DELETE)

TRUNCATE TABLE table_name;

TRUNCATE TABLE staff;


 1.5 DROP (remove structure and data)
DROP TABLE table_name;

DROP TABLE staff;
DROP DATABASE company_db;


(Use delete if you want to delete a row based on condition / complete table - It is part of DML)

 2) DML — Insert/Change/Delete Data

 Let’s recreate the table for the examples:


CREATE DATABASE company_db;
USE company_db;

CREATE TABLE employees (
    id INT PRIMARY KEY AUTO_INCREMENT,
    first_name VARCHAR(50) NOT NULL,
    last_name  VARCHAR(50) NOT NULL,
    email      VARCHAR(100) UNIQUE,
    hire_date  DATE NOT NULL,
    salary     DECIMAL(12,2) NOT NULL DEFAULT 0.00,
    dept       VARCHAR(50) DEFAULT 'General'
);


 2.1 INSERT (single & multiple rows)

INSERT INTO table_name (column1, column2, ...)
VALUES (value1, value2, ...);

--all col
INSERT INTO table_name
VALUES (value1, value2, value3);

--multiple rows insert 
INSERT INTO table_name (col1, col2)
VALUES 
(valueA1, valueA2),
(valueB1, valueB2),
(valueC1, valueC2);


-- Single row
INSERT INTO employees (first_name, last_name, email, hire_date, salary, dept)
VALUES ('Anita', 'Rao', 'anita.rao@example.com', '2024-01-10', 65000.00, 'HR');

-- Multiple rows
INSERT INTO employees (first_name, last_name, email, hire_date, salary, dept)
VALUES
('Vikram', 'Shah', 'vikram.shah@example.com', '2023-11-05', 80000.00, 'Engineering'),
('Meera',  'mithun', 'meera.mithun@example.com',  '2025-02-01', 72000.00, 'Finance');


 MySQL tip: `REPLACE INTO` inserts a row or overwrites (deletes + inserts) if a unique/primary key conflict occurs.


REPLACE INTO employees (id, first_name, last_name, email, hire_date, salary, dept)
VALUES (1, 'Anita', 'Rao', 'anita.rao@example.com', '2024-01-10', 70000.00, 'HR');


 2.2 UPDATE (change existing rows)

UPDATE table_name
SET column1 = value1, column2 = value2
WHERE condition;

UPDATE table_name
SET column1 = value1;

-- Give a 10% raise to Engineering
UPDATE employees
SET salary = 50000
WHERE dept = 'Engineering';


 2.3 DELETE (remove rows)

DELETE FROM table_name
WHERE condition;

DELETE FROM table_name;


-- Delete employees hired before 2024
DELETE FROM employees
WHERE hire_date < '2024-01-01';


 2.4 Transactions (TCL with DML)

 Use transactions to ensure multiple changes succeed or fail together.

 3.1 Basic SELECT

-- All columns
SELECT  * FROM employees;

SELECT column1, column2
FROM table_name;

SELECT *
FROM table_name
WHERE condition;

-- with limit
SELECT *
FROM table_name
LIMIT 5;

--by sorting 
SELECT *
FROM table_name
ORDER BY column_name ASC|DESC;


-- Specific columns
SELECT id, first_name, last_name, dept FROM employees;


-----------------------------------------------------------------------------------------------------------------------------

 3.2 Filtering (WHERE), Ordering, Limiting

-- Filter
SELECT first_name, last_name, salary
FROM employees
WHERE dept = 'Engineering' AND salary = 75000;

-- Sort by salary descending
SELECT first_name, last_name, salary
FROM employees
ORDER BY salary DESC;

-- Limit to top 3
SELECT first_name, last_name, salary
FROM employees
ORDER BY salary DESC
LIMIT 3;

 3.3 Aggregations (COUNT, SUM, AVG, MIN, MAX) + GROUP BY + HAVING


* GROUP BY
GROUP BY groups rows that have the same values in specified columns and allows you to run aggregate functions like:

COUNT()
SUM()
AVG()
MAX()
MIN()


* HAVING
HAVING is used to filter groups after aggregation.

WHERE → filters rows before grouping
HAVING → filters groups after grouping


-- Average salary by department
SELECT dept, AVG(salary) AS avg_salary
FROM employees
GROUP BY dept;

orginal table  --> where sal>10000
dept	 sal 		dept 	sal

d1	10000		d2	20000
d2	20000		d2	20000

d2	20000
d1	10000
--------------

vir table --> filter avg(sal) >10000

dept	 sal    dept	Avg_sal	 	dept	Avg(sal)
	
d1	10000   d1	10000		d2	20000
d1	10000

d2	20000   d2      20000
d2	20000

----------------------------------
-- 

Having --> filtering based on condition 

Departments with avg salary above 70k
SELECT dept, AVG(salary) AS avg_salary
FROM employees
GROUP BY dept
HAVING AVG(salary)  70000;


--------------------------------------------------------------------------------------------------------

 3.4 Joins 


-- Create a departments table
CREATE TABLE departments (
    dept VARCHAR(50) PRIMARY KEY,
    location VARCHAR(50) NOT NULL
);

INSERT INTO departments (dept, location) VALUES
('Engineering', 'Bengaluru'),
('Finance', 'Chennai'),
('HR', 'Coimbatore'),
('General', 'Hyderabad');

-- INNER JOIN example: show employee with department location
SELECT e.first_name, e.last_name, e.dept, d.location
FROM employees AS e
JOIN departments AS d
  ON e.dept = d.dept;

-- LEFT JOIN example: include employees even if dept missing in departments
SELECT e.first_name, e.last_name, e.dept, d.location
FROM employees AS e
LEFT JOIN departments AS d
  ON e.dept = d.dept;


 3.5 Useful MySQL functions in SELECT


-- String concatenation
SELECT CONCAT(first_name, ' ', last_name) AS full_name FROM employees;

-- Date comparisons
SELECT first_name, hire_date
FROM employees
WHERE hire_date BETWEEN '2024-01-01' AND '2025-12-31';

-- Conditional logic
SELECT first_name, salary,
       CASE
         WHEN salary = 80000 THEN 'High'
         WHEN salary = 60000 THEN 'Medium'
         ELSE 'Low'
       END AS salary_band
FROM employees;


-----------------------------------------------------------------------------------------------------------------------

START TRANSACTION;

UPDATE employees
SET salary = salary + 5000
WHERE dept = 'Finance';

-- Oops? Roll it back:
-- ROLLBACK;

-- If all looks good:
COMMIT;
